## Step 2: HAR Parser Module

### `src/har_parser.py`
```python
"""
HAR File Parser - Extracts endpoints, auth tokens, headers, and request patterns
from recorded HAR files for Salesforce Assist and Tenant portals.
"""

import json
import re
from pathlib import Path
from urllib.parse import urlparse, parse_qs, urlencode
from dataclasses import dataclass, field
from typing import Optional
from datetime import datetime


@dataclass
class ParsedEndpoint:
    """Represents a single API endpoint extracted from HAR."""
    url: str
    method: str
    status_code: int
    request_headers: dict
    response_headers: dict
    request_body: Optional[str] = None
    response_body: Optional[str] = None
    query_params: dict = field(default_factory=dict)
    path_params: dict = field(default_factory=dict)
    content_type: str = ""
    auth_type: str = ""
    auth_token: str = ""
    cookies: list = field(default_factory=list)
    is_api: bool = False
    is_static: bool = False
    object_type: str = ""
    record_id: str = ""
    api_version: str = ""
    raw_request: dict = field(default_factory=dict)
    raw_response: dict = field(default_factory=dict)


@dataclass
class HarAnalysis:
    """Complete analysis of a HAR file."""
    portal_name: str
    base_url: str
    endpoints: list[ParsedEndpoint] = field(default_factory=list)
    auth_tokens: dict = field(default_factory=dict)
    cookies: list = field(default_factory=list)
    api_endpoints: list[ParsedEndpoint] = field(default_factory=list)
    static_resources: list[ParsedEndpoint] = field(default_factory=list)
    api_versions: set = field(default_factory=set)
    sobjects_discovered: set = field(default_factory=set)
    apex_endpoints: list[ParsedEndpoint] = field(default_factory=list)
    unique_methods: set = field(default_factory=set)
    error_responses: list[ParsedEndpoint] = field(default_factory=list)
    parsed_at: str = field(default_factory=lambda: datetime.utcnow().isoformat())


class HarParser:
    """Parses HAR files and extracts structured endpoint data."""

    # Salesforce-specific patterns
    SF_API_PATTERN = re.compile(
        r'/services/data/v(\d+\.\d+)/(.*)'
    )
    SF_APEX_PATTERN = re.compile(
        r'/services/apexrest/(.*)'
    )
    SF_SOBJECT_PATTERN = re.compile(
        r'/services/data/v[\d.]+/sobjects/(\w+)(?:/(\w+))?'
    )
    SF_QUERY_PATTERN = re.compile(
        r'/services/data/v[\d.]+/query/?'
    )
    STATIC_EXTENSIONS = {'.js', '.css', '.png', '.jpg', '.gif', '.svg', '.ico', '.woff', '.woff2'}
    SENSITIVE_HEADERS = {'authorization', 'cookie', 'x-sfdc-session', 'set-cookie'}

    def __init__(self, har_path: str, portal_name: str):
        self.har_path = Path(har_path)
        self.portal_name = portal_name
        self._raw_har = None

    def parse(self) -> HarAnalysis:
        """Main entry point - parse HAR file and return structured analysis."""
        print(f"[HAR Parser] Parsing: {self.har_path} for portal: {self.portal_name}")

        with open(self.har_path, 'r', encoding='utf-8') as f:
            self._raw_har = json.load(f)

        entries = self._raw_har.get('log', {}).get('entries', [])
        print(f"[HAR Parser] Found {len(entries)} entries in HAR file")

        analysis = HarAnalysis(portal_name=self.portal_name, base_url=self._extract_base_url(entries))

        for entry in entries:
            endpoint = self._parse_entry(entry)
            if endpoint is None:
                continue

            analysis.endpoints.append(endpoint)
            self._categorize_endpoint(endpoint, analysis)

        # Extract auth info
        analysis.auth_tokens = self._extract_auth_tokens(analysis.endpoints)
        analysis.cookies = self._extract_all_cookies(analysis.endpoints)

        print(f"[HAR Parser] Analysis complete:")
        print(f"  - Total endpoints: {len(analysis.endpoints)}")
        print(f"  - API endpoints: {len(analysis.api_endpoints)}")
        print(f"  - Apex endpoints: {len(analysis.apex_endpoints)}")
        print(f"  - SObjects discovered: {analysis.sobjects_discovered}")
        print(f"  - API versions: {analysis.api_versions}")
        print(f"  - Error responses: {len(analysis.error_responses)}")

        return analysis

    def _extract_base_url(self, entries: list) -> str:
        """Extract the base URL from HAR entries."""
        for entry in entries:
            url = entry.get('request', {}).get('url', '')
            parsed = urlparse(url)
            if parsed.scheme and parsed.netloc:
                return f"{parsed.scheme}://{parsed.netloc}"
        return ""

    def _parse_entry(self, entry: dict) -> Optional[ParsedEndpoint]:
        """Parse a single HAR entry into a ParsedEndpoint."""
        request = entry.get('request', {})
        response = entry.get('response', {})

        url = request.get('url', '')
        parsed_url = urlparse(url)

        # Skip data URIs and about:blank
        if not parsed_url.scheme or parsed_url.scheme in ('data', 'about'):
            return None

        # Determine if static resource
        path = parsed_url.path
        ext = Path(path).suffix.lower()
        is_static = ext in self.STATIC_EXTENSIONS

        # Parse request headers
        req_headers = {h['name']: h['value'] for h in request.get('headers', [])}

        # Parse response headers
        resp_headers = {h['name']: h['value'] for h in response.get('headers', [])}

        # Get request body
        req_body = None
        post_data = request.get('postData', {})
        if post_data:
            req_body = post_data.get('text', '')

        # Get response body
        resp_body = None
        resp_content = response.get('content', {})
        if resp_content:
            resp_body = resp_content.get('text', '')

        # Determine auth type
        auth_type = ""
        auth_token = ""
        auth_header = req_headers.get('Authorization', '')
        if auth_header.startswith('Bearer '):
            auth_type = "Bearer"
            auth_token = auth_header[7:]
        elif auth_header.startswith('OAuth '):
            auth_type = "OAuth"
            auth_token = auth_header[6:]
        elif 'X-SFDC-Session' in req_headers:
            auth_type = "SFDC-Session"
            auth_token = req_headers['X-SFDC-Session']

        # Parse cookies
        cookies = []
        for cookie in request.get('cookies', []):
            cookies.append({
                'name': cookie.get('name', ''),
                'value': cookie.get('value', ''),
                'httpOnly': cookie.get('httpOnly', False),
                'secure': cookie.get('secure', False)
            })

        # Check if Salesforce API
        is_api = bool(self.SF_API_PATTERN.search(path) or self.SF_APEX_PATTERN.search(path))

        # Extract SObject info
        object_type = ""
        record_id = ""
        api_version = ""

        sobj_match = self.SF_SOBJECT_PATTERN.search(path)
        if sobj_match:
            object_type = sobj_match.group(1)
            record_id = sobj_match.group(2) or ""

        api_match = self.SF_API_PATTERN.search(path)
        if api_match:
            api_version = api_match.group(1)

        return ParsedEndpoint(
            url=url,
            method=request.get('method', 'GET'),
            status_code=response.get('status', 0),
            request_headers=req_headers,
            response_headers=resp_headers,
            request_body=req_body,
            response_body=resp_body,
            query_params=parse_qs(parsed_url.query),
            content_type=resp_headers.get('Content-Type', ''),
            auth_type=auth_type,
            auth_token=auth_token,
            cookies=cookies,
            is_api=is_api,
            is_static=is_static,
            object_type=object_type,
            record_id=record_id,
            api_version=api_version,
            raw_request=request,
            raw_response=response,
        )

    def _categorize_endpoint(self, endpoint: ParsedEndpoint, analysis: HarAnalysis):
        """Categorize an endpoint into the appropriate collection."""
        if endpoint.is_static:
            analysis.static_resources.append(endpoint)
            return

        analysis.unique_methods.add(endpoint.method)

        if endpoint.status_code >= 400:
            analysis.error_responses.append(endpoint)

        if endpoint.is_api:
            analysis.api_endpoints.append(endpoint)

            if endpoint.api_version:
                analysis.api_versions.add(endpoint.api_version)

            if endpoint.object_type:
                analysis.sobjects_discovered.add(endpoint.object_type)

        if self.SF_APEX_PATTERN.search(urlparse(endpoint.url).path):
            analysis.apex_endpoints.append(endpoint)

    def _extract_auth_tokens(self, endpoints: list[ParsedEndpoint]) -> dict:
        """Extract all unique auth tokens from endpoints."""
        tokens = {}
        for ep in endpoints:
            if ep.auth_type and ep.auth_token:
                key = f"{ep.auth_type}_token"
                if key not in tokens:
                    tokens[key] = ep.auth_token
        return tokens

    def _extract_all_cookies(self, endpoints: list[ParsedEndpoint]) -> list:
        """Extract unique cookies from all endpoints."""
        seen = set()
        cookies = []
        for ep in endpoints:
            for cookie in ep.cookies:
                cookie_key = cookie['name']
                if cookie_key not in seen:
                    seen.add(cookie_key)
                    cookies.append(cookie)
        return cookies
```

```python
"""
Evidence Collector - Captures HTTP request/response pairs, screenshots,
and structures proof for each security test case.

class EvidenceCollector:
    """Collects and stores evidence for security test executions."""

    # Patterns for sensitive data redaction
    REDACT_PATTERNS = [
        (r'(Bearer\s+)[\w\-\.]+', r'\1[REDACTED_TOKEN]'),
        (r'(session_id=)[^&\s]+', r'\1[REDACTED]'),
        (r'("access_token"\s*:\s*)"[^"]+"', r'\1"[REDACTED]"'),
        (r'("refresh_token"\s*:\s*)"[^"]+"', r'\1"[REDACTED]"'),
        (r'(sid=)[^;]+', r'\1[REDACTED]'),
    ]

    def __init__(self, output_dir: str, redact_sensitive: bool = True):
        self.output_dir = Path(output_dir)
        self.redact_sensitive = redact_sensitive
        self.run_id = datetime.utcnow().strftime("%Y%m%d_%H%M%S")
        self.run_dir = self.output_dir / self.run_id

        # Create directory structure
        (self.run_dir / "requests").mkdir(parents=True, exist_ok=True)
        (self.run_dir / "responses").mkdir(parents=True, exist_ok=True)
        (self.run_dir / "screenshots").mkdir(parents=True, exist_ok=True)
        (self.run_dir / "curl_commands").mkdir(parents=True, exist_ok=True)

    def send_and_capture(
        self,
        test_case_id: str,
        test_case_name: str,
        method: str,
        url: str,
        headers: dict = None,
        body: str = None,
        timeout: int = 30,
        verify_ssl: bool = True,
        notes: str = ""
    ) -> EvidenceItem:
        """
        Send an HTTP request and capture full evidence including
        request details, response details, and timing.
        """
        headers = headers or {}
        timestamp = datetime.utcnow().isoformat()

        try:
            # Execute request with timing
            start_time = datetime.utcnow()
            response = requests.request(
                method=method,
                url=url,
                headers=headers,
                data=body,
                timeout=timeout,
                verify=verify_ssl,
                allow_redirects=True
            )
            end_time = datetime.utcnow()
            response_time_ms = (end_time - start_time).total_seconds() * 1000

            # Get response body
            try:
                resp_body = response.text
            except Exception:
                resp_body = "[Could not decode response body]"

            # Generate curl command
            curl_cmd = self._generate_curl(method, url, headers, body)

            # Save raw request/response to files
            request_filename = f"{test_case_id}_{self._hash_url(url)}_req.json"
            response_filename = f"{test_case_id}_{self._hash_url(url)}_resp.json"

            self._save_json(
                self.run_dir / "requests" / request_filename,
                {
                    "method": method,
                    "url": url,
                    "headers": self._maybe_redact_dict(headers),
                    "body": self._maybe_redact(body),
                    "timestamp": timestamp
                }
            )

            self._save_json(
                self.run_dir / "responses" / response_filename,
                {
                    "status_code": response.status_code,
                    "headers": dict(response.headers),
                    "body": self._maybe_redact(resp_body),
                    "response_time_ms": response_time_ms,
                    "url": response.url,
                    "timestamp": timestamp
                }
            )

            # Save curl command
            curl_filename = f"{test_case_id}_{self._hash_url(url)}.txt"
            (self.run_dir / "curl_commands" / curl_filename).write_text(curl_cmd)

            return EvidenceItem(
                timestamp=timestamp,
                test_case_id=test_case_id,
                test_case_name=test_case_name,
                request_method=method,
                request_url=url,
                request_headers=self._maybe_redact_dict(headers),
                request_body=self._maybe_redact(body),
                response_status_code=response.status_code,
                response_headers=dict(response.headers),
                response_body=self._maybe_redact(resp_body[:10000]),  # Cap at 10KB
                response_time_ms=response_time_ms,
                notes=notes,
                curl_command=curl_cmd
            )

        except requests.exceptions.Timeout:
            return EvidenceItem(
                timestamp=timestamp,
                test_case_id=test_case_id,
                test_case_name=test_case_name,
                request_method=method,
                request_url=url,
                request_headers=self._maybe_redact_dict(headers),
                request_body=self._maybe_redact(body),
                response_status_code=0,
                response_headers={},
                response_body="[TIMEOUT]",
                response_time_ms=timeout * 1000,
                notes=f"Request timed out after {timeout}s",
                curl_command=self._generate_curl(method, url, headers, body)
            )

        except requests.exceptions.ConnectionError as e:
            return EvidenceItem(
                timestamp=timestamp,
                test_case_id=test_case_id,
                test_case_name=test_case_name,
                request_method=method,
                request_url=url,
                request_headers=self._maybe_redact_dict(headers),
                request_body=self._maybe_redact(body),
                response_status_code=0,
                response_headers={},
                response_body=f"[CONNECTION ERROR] {str(e)}",
                response_time_ms=0,
                notes=f"Connection error: {str(e)}",
                curl_command=self._generate_curl(method, url, headers, body)
            )

        except Exception as e:
            return EvidenceItem(
                timestamp=timestamp,
                test_case_id=test_case_id,
                test_case_name=test_case_name,
                request_method=method,
                request_url=url,
                request_headers=self._maybe_redact_dict(headers),
                request_body=self._maybe_redact(body),
                response_status_code=0,
                response_headers={},
                response_body=f"[ERROR] {str(e)}",
                response_time_ms=0,
                notes=f"Unexpected error: {str(e)}",
                curl_command=self._generate_curl(method, url, headers, body)
            )

    def save_screenshot(self, test_case_id: str, url: str, page=None) -> Optional[str]:
        """Save a Playwright screenshot for visual evidence."""
        if page is None:
            return None

        try:
            filename = f"{test_case_id}_{self._hash_url(url)}.png"
            filepath = self.run_dir / "screenshots" / filename
            page.screenshot(path=str(filepath), full_page=True)
            return str(filepath)
        except Exception as e:
            print(f"[Evidence] Screenshot failed: {e}")
            return None

    def _generate_curl(self, method: str, url: str, headers: dict, body: str = None) -> str:
        """Generate a reproducible curl command from request details."""
        parts = [f"curl -X {method}"]

        for key, value in headers.items():
            safe_value = self._maybe_redact(value) if self.redact_sensitive else value
            parts.append(f"  -H '{key}: {safe_value}'")

        if body:
            safe_body = self._maybe_redact(body) if self.redact_sensitive else body
            parts.append(f"  -d '{safe_body}'")

        parts.append(f"  '{url}'")
        return " \\\n".join(parts)

    def _maybe_redact(self, text: Optional[str]) -> Optional[str]:
        """Redact sensitive patterns from text if enabled."""
        if text is None or not self.redact_sensitive:
            return text
        result = text
        for pattern, replacement in self.REDACT_PATTERNS:
            result = re.sub(pattern, replacement, result, flags=re.IGNORECASE)
        return result

    def _maybe_redact_dict(self, d: dict) -> dict:
        """Redact sensitive values in a dictionary."""
        if not self.redact_sensitive:
            return d
        sensitive_keys = {'authorization', 'cookie', 'x-sfdc-session', 'set-cookie'}
        result = {}
        for k, v in d.items():
            if k.lower() in sensitive_keys:
                result[k] = "[REDACTED]"
            else:
                result[k] = v
        return result

    def _hash_url(self, url: str) -> str:
        """Create a short hash from a URL for filenames."""
        return hashlib.md5(url.encode()).hexdigest()[:8]

    def _save_json(self, filepath: Path, data: dict):
        """Save data as formatted JSON."""
        with open(filepath, 'w', encoding='utf-8') as f:
            json.dump(data, f, indent=2, ensure_ascii=False, default=str)
```

```python
"""
Abstract base class for all security tests. Provides common functionality
for test execution, evidence collection, and result determination.
"""

import time
import json
import re
from abc import ABC, abstractmethod
from typing import Optional
from src.evidence_collector import EvidenceCollector, TestResult, EvidenceItem
from src.har_parser import HarAnalysis, ParsedEndpoint


class BaseSecurityTest(ABC):
    """Base class all security tests must inherit from."""

    def __init__(
        self,
        har_analysis: HarAnalysis,
        evidence_collector: EvidenceCollector,
        config: dict,
        base_url: str,
        auth_headers: dict
    ):
        self.har = har_analysis
        self.evidence = evidence_collector
        self.config = config
        self.base_url = base_url
        self.auth_headers = auth_headers
        self.results: list[TestResult] = []

    @abstractmethod
    def run(self) -> list[TestResult]:
        """Execute all test cases in this test suite. Must return list of TestResult."""
        pass

    def create_result(
        self,
        test_case_id: str,
        test_case_name: str,
        suite_id: str,
        suite_name: str,
        owasp_api: str,
        owasp_web: str,
        applicable: bool = True,
        applicability_reason: str = "Test is applicable to discovered endpoints"
    ) -> TestResult:
        """Factory method for creating a TestResult."""
        return TestResult(
            test_case_id=test_case_id,
            test_case_name=test_case_name,
            suite_id=suite_id,
            suite_name=suite_name,
            owasp_api_category=owasp_api,
            owasp_web_category=owasp_web,
            status="NOT_FINDING",
            severity="INFO",
            confidence="HIGH",
            description="",
            applicable=applicable,
            applicability_reason=applicability_reason
        )

    def mark_finding(
        self,
        result: TestResult,
        severity: str,
        confidence: str,
        description: str,
        recommendation: str = ""
    ):
        """Mark a test result as a finding."""
        result.status = "FINDING"
        result.severity = severity
        result.confidence = confidence
        result.description = description
        result.recommendation = recommendation

    def mark_na(self, result: TestResult, reason: str):
        """Mark a test result as Not Applicable."""
        result.status = "NA"
        result.applicable = False
        result.applicability_reason = reason

    def send_request(
        self,
        test_case_id: str,
        test_case_name: str,
        method: str,
        url: str,
        headers: dict = None,
        body: str = None,
        notes: str = ""
    ) -> EvidenceItem:
        """Send a request through the evidence collector."""
        merged_headers = {**self.auth_headers, **(headers or {})}
        return self.evidence.send_and_capture(
            test_case_id=test_case_id,
            test_case_name=test_case_name,
            method=method,
            url=url,
            headers=merged_headers,
            body=body,
            notes=notes
        )

    def get_api_endpoints(self) -> list[ParsedEndpoint]:
        """Get all API endpoints from HAR analysis."""
        return self.har.api_endpoints

    def get_endpoints_by_method(self, method: str) -> list[ParsedEndpoint]:
        """Filter endpoints by HTTP method."""
        return [ep for ep in self.har.api_endpoints if ep.method.upper() == method.upper()]

    def get_endpoints_with_object_id(self) -> list[ParsedEndpoint]:
        """Get endpoints that contain SObject IDs in the URL."""
        return [ep for ep in self.har.api_endpoints if ep.record_id]

    def get_writable_endpoints(self) -> list[ParsedEndpoint]:
        """Get endpoints that support write operations."""
        return [ep for ep in self.har.api_endpoints if ep.method.upper() in ('POST', 'PUT', 'PATCH')]

    def get_query_endpoints(self) -> list[ParsedEndpoint]:
        """Get endpoints that support SOQL queries."""
        return [ep for ep in self.har.api_endpoints if '/query' in ep.url]

    def has_url_params(self, endpoint: ParsedEndpoint) -> bool:
        """Check if endpoint has URL query parameters."""
        return bool(endpoint.query_params)

    def extract_ids_from_url(self, url: str) -> list[str]:
        """Extract Salesforce record IDs (15 or 18 char) from URL."""
        # Salesforce IDs: 15 or 18 alphanumeric characters
        pattern = r'\b([a-zA-Z0-9]{15}|[a-zA-Z0-9]{18})\b'
        return re.findall(pattern, url)

    def replace_id_in_url(self, url: str, original_id: str, new_id: str) -> str:
        """Replace a record ID in a URL with a different one."""
        return url.replace(original_id, new_id)

    def generate_fake_sf_id(self, prefix: str = "001") -> str:
        """Generate a plausible but likely non-existent Salesforce ID."""
        import random
        import string
        chars = string.ascii_letters + string.digits
        return prefix + ''.join(random.choice(chars) for _ in range(12)) + "XXX"
```

```python
"""
API1:2023 - Broken Object Level Authorization (BOLA/IDOR)
Tests for IDOR vulnerabilities in Salesforce REST API endpoints.
"""

import re
from tests.base_test import BaseSecurityTest
from src.evidence_collector import TestResult


class BOLA_Test(BaseSecurityTest):

    SUITE_ID = "API1-BOLA"
    SUITE_NAME = "Broken Object Level Authorization"
    OWASP_API = "API1:2023"
    OWASP_WEB = "A01:2021"

    def run(self) -> list[TestResult]:
        results = []

        results.append(self._test_idor_sobject_access())
        results.append(self._test_idor_apex_rest())
        results.append(self._test_idor_composite_api())
        results.append(self._test_cross_tenant_isolation())
        results.append(self._test_bulk_api_idor())

        return results

    def _test_idor_sobject_access(self) -> TestResult:
        """
        Test: Replace object IDs in SObject REST API URLs with IDs of
        records that should not be accessible to the current user.

        Methodology:
        1. Identify all GET endpoints with SObject IDs from HAR
        2. For each, substitute the ID with a fabricated/different ID
        3. Compare response status codes and body content
        4. If 200 OK with data → FINDING (BOLA confirmed)
        5. If 403/404 → NOT_FINDING (access control working)
        """
        result = self.create_result(
            "API1-001", "IDOR on Record Access via REST API",
            self.SUITE_ID, self.SUITE_NAME, self.OWASP_API, self.OWASP_WEB
        )

        id_endpoints = self.get_endpoints_with_object_id()

        if not id_endpoints:
            self.mark_na(result, "No endpoints with SObject IDs found in HAR file")
            return result

        findings_count = 0
        tested_count = 0

        for endpoint in id_endpoints[:10]:  # Limit to first 10 to avoid excessive requests
            if endpoint.method.upper() != 'GET':
                continue

            original_id = endpoint.record_id
            if not original_id:
                continue

            tested_count += 1

            # Test 1: Replace with a fake ID
            fake_id = self.generate_fake_sf_id(original_id[:3])
            modified_url = self.replace_id_in_url(endpoint.url, original_id, fake_id)

            evidence_fake = self.send_request(
                "API1-001", result.test_case_name,
                "GET", modified_url,
                notes=f"Fake ID substitution: {original_id} → {fake_id}"
            )
            result.add_evidence(evidence_fake)

            # Test 2: Replace with a different but real-looking ID pattern
            alt_id = self._mutate_id(original_id)
            alt_url = self.replace_id_in_url(endpoint.url, original_id, alt_id)

            evidence_alt = self.send_request(
                "API1-001", result.test_case_name,
                "GET", alt_url,
                notes=f"Mutated ID substitution: {original_id} → {alt_id}"
            )
            result.add_evidence(evidence_alt)

            # Analyze: Did the API return 200 with data for unauthorized access?
            for ev in [evidence_fake, evidence_alt]:
                if ev.response_status_code == 200:
                    try:
                        body = eval(ev.response_body) if ev.response_body else {}
                        if isinstance(body, dict) and body.get('Id'):
                            findings_count += 1
                    except Exception:
                        pass

        if findings_count > 0:
            self.mark_finding(
                result, "CRITICAL", "HIGH",
                f"BOLA detected: {findings_count}/{tested_count} endpoints returned data "
                f"for substituted object IDs. Object-level access controls are not enforced.",
                "Implement proper record-level sharing checks. Use Salesforce sharing rules "
                "and ensure Apex controllers enforce WITH SHARING. Validate record ownership "
                "before returning data."
            )
        else:
            result.status = "NOT_FINDING"
            result.severity = "INFO"
            result.description = (
                f"Tested {tested_count} endpoints with ID substitution. "
                "All returned appropriate error responses (403/404). "
                "Object-level authorization appears to be enforced."
            )

        return result

    def _test_idor_apex_rest(self) -> TestResult:
        """Test custom Apex REST endpoints for BOLA."""
        result = self.create_result(
            "API1-002", "IDOR on Custom Apex REST Endpoints",
            self.SUITE_ID, self.SUITE_NAME, self.OWASP_API, self.OWASP_WEB
        )

        apex_endpoints = self.har.apex_endpoints

        if not apex_endpoints:
            self.mark_na(result, "No Apex REST endpoints found in HAR")
            return result

        findings_count = 0

        for endpoint in apex_endpoints[:5]:
            ids_in_url = self.extract_ids_from_url(endpoint.url)
            if not ids_in_url:
                continue

            for sf_id in ids_in_url:
                fake_id = self.generate_fake_sf_id(sf_id[:3])
                modified_url = self.replace_id_in_url(endpoint.url, sf_id, fake_id)

                evidence = self.send_request(
                    "API1-002", result.test_case_name,
                    endpoint.method, modified_url,
                    notes=f"Apex REST IDOR test: {sf_id} → {fake_id}"
                )
                result.add_evidence(evidence)

                if evidence.response_status_code == 200 and evidence.response_body:
                    # Check if response contains actual record data
                    if len(evidence.response_body) > 100:
                        findings_count += 1

        if findings_count > 0:
            self.mark_finding(
                result, "HIGH", "HIGH",
                f"Apex REST endpoints returned data for substituted IDs ({findings_count} cases). "
                "Custom REST endpoints may not enforce object-level authorization.",
                "Review Apex REST controllers for proper sharing enforcement. "
                "Ensure 'with sharing' keyword is used in controller classes."
            )
        else:
            result.status = "NOT_FINDING"
            result.description = "Apex REST endpoints properly handle ID substitution."

        return result

    def _test_idor_composite_api(self) -> TestResult:
        """Test Salesforce Composite API for BOLA."""
        result = self.create_result(
            "API1-003", "BOLA via Composite API",
            self.SUITE_ID, self.SUITE_NAME, self.OWASP_API, self.OWASP_WEB
        )

        # Check if composite API endpoint was seen in HAR
        composite_seen = any('/composite' in ep.url for ep in self.har.api_endpoints)

        if not composite_seen and not self.har.sobjects_discovered:
            self.mark_na(result, "Composite API not used and no SObjects discovered")
            return result

        # Build a composite request with substituted IDs
        version = list(self.har.api_versions)[0] if self.har.api_versions else "58.0"
        composite_url = f"{self.base_url}/services/data/v{version}/composite"

        # Create subrequests targeting different objects with fake IDs
        subrequests = []
        for sobj in list(self.har.sobjects_discovered)[:3]:
            fake_id = self.generate_fake_sf_id("001")
```

```python
    def _test_jwt_weaknesses(self) -> TestResult:
        """Analyze JWT tokens for common weaknesses."""
        result = self.create_result(
            "API2-004", "Weak JWT Implementation",
            self.SUITE_ID, self.SUITE_NAME, self.OWASP_API, self.OWASP_WEB
        )

        jwt_tokens = []
        for token_type, token_value in self.har.auth_tokens.items():
            if token_value.count('.') == 2:  # Looks like JWT
                jwt_tokens.append((token_type, token_value))

        if not jwt_tokens:
            self.mark_na(result, "No JWT tokens found in HAR file")
            return result

        issues = []

        for token_type, token in jwt_tokens:
            try:
                parts = token.split('.')
                # Decode header
                header_b64 = parts[0] + '=' * (4 - len(parts[0]) % 4)
                header = json.loads(base64.urlsafe_b64decode(header_b64))

                # Check algorithm
                alg = header.get('alg', '').upper()
                if alg == 'NONE':
                    issues.append(f"JWT uses 'none' algorithm ({token_type})")
                if alg in ('HS256', 'HS384', 'HS512'):
                    issues.append(f"JWT uses symmetric algorithm {alg} - verify key strength ({token_type})")

                # Decode payload
                payload_b64 = parts[1] + '=' * (4 - len(parts[1]) % 4)
                payload = json.loads(base64.urlsafe_b64decode(payload_b64))

                # Check for expiration
                if 'exp' not in payload:
                    issues.append(f"JWT missing expiration claim ({token_type})")
                elif payload['exp'] > time.time() + 86400 * 30:
                    issues.append(f"JWT has excessively long expiration ({token_type})")

                # Check for sensitive data in payload
                sensitive_claims = ['password', 'secret', 'ssn', 'credit_card']
                for claim in sensitive_claims:
                    if claim in payload:
                        issues.append(f"JWT contains sensitive data: {claim} ({token_type})")

            except Exception as e:
                issues.append(f"Could not parse JWT ({token_type}): {str(e)}")

        if issues:
            self.mark_finding(
                result, "HIGH", "HIGH",
                "JWT weaknesses detected:\n" + "\n".join(f"  - {i}" for i in issues),
                "Use RS256 or ES256 algorithms. Set reasonable expiration times. "
                "Never include sensitive data in JWT payload. Implement token rotation."
            )
        else:
            result.status = "NOT_FINDING"
            result.description = "JWT tokens appear to follow best practices."

        return result
```

```python
"""
```

```python
manages results, and coordinates evidence collection.
"""

import time
import yaml
from pathlib import Path
from datetime import datetime
from typing import Optional

class TestEngine:
    """Main test execution engine."""

    # Registry of all available test suites
    TEST_REGISTRY = {
        "API1-BOLA": BOLA_Test,
        "API2-AUTH": BrokenAuthTest,
        "WEB-A03-INJECTION": InjectionTest,
        "API8-MISCONFIG": MisconfigTest,

        # "API3-EXCESSIVE-DATA": ExcessiveDataTest,

        # "API7-SSRF": SSRFTest,
    }

    def __init__(self, config_path: str = "config/settings.yaml"):
        with open(config_path, 'r') as f:
            self.config = yaml.safe_load(f)

        self.evidence_collector = EvidenceCollector(
            output_dir=self.config['output']['evidence_dir'],
            redact_sensitive=self.config['output'].get('redact_sensitive', True)
        )
        self.all_results = []
        self.execution_start = None
        self.execution_end = None

    def run_full_assessment(self):
        """Run the complete security assessment for all configured targets."""
        self.execution_start = datetime.utcnow()
        print("=" * 80)
        print(f"  SECURITY ASSESSMENT STARTED: {self.execution_start.isoformat()}")
        print(f"  Project: {self.config['project']['name']}")
        print("=" * 80)

        for target_key, target_config in self.config['targets'].items():
            print(f"\n{'='*60}")
            print(f"  TARGET: {target_config['name']}")
            print(f"  Base URL: {target_config['base_url']}")
            print(f"{'='*60}\n")

            self._assess_target(target_key, target_config)

        self.execution_end = datetime.utcnow()
        duration = (self.execution_end - self.execution_start).total_seconds()

        print(f"\n{'='*80}")
        print(f"  ASSESSMENT COMPLETE")
        print(f"  Duration: {duration:.1f} seconds")
        print(f"  Total Tests: {len(self.all_results)}")
        print(f"  Findings: {sum(1 for r in self.all_results if r.status == 'FINDING')}")
        print(f"  Not Findings: {sum(1 for r in self.all_results if r.status == 'NOT_FINDING')}")
        print(f"  N/A: {sum(1 for r in self.all_results if r.status == 'NA')}")
        print(f"{'='*80}\n")

        # Generate reports
        self._generate_reports()

        return self.all_results

    def _assess_target(self, target_key: str, target_config: dict):
        """Assess a single target (portal)."""

        # Step 1: Parse HAR file
        print(f"[{target_key}] Step 1: Parsing HAR file...")
        parser = HarParser(target_config['har_file'], target_config['name'])
        har_analysis = parser.parse()

        # Step 2: Extract auth headers
        auth_headers = self._build_auth_headers(har_analysis)
        print(f"[{target_key}] Auth type detected: {list(har_analysis.auth_tokens.keys())}")

        # Step 3: Run each test suite
        for suite_id, test_class in self.TEST_REGISTRY.items():
            print(f"\n[{target_key}] Running: {suite_id} - {test_class.SUITE_NAME}...")
            suite_start = time.time()

            try:
                test_instance = test_class(
                    har_analysis=har_analysis,
                    evidence_collector=self.evidence_collector,
                    config=self.config,
                    base_url=target_config['base_url'],
                    auth_headers=auth_headers
                )

                results = test_instance.run()

                for result in results:
                    result.execution_time_seconds = time.time() - suite_start
                    self.all_results.append(result)

                    # Print result summary
                    status_icon = {
                        "FINDING": "🔴",
                        "NOT_FINDING": "🟢",
                        "NA": "⚪",
                        "ERROR": "🟡"
                    }.get(result.status, "❓")

                    print(f"  {status_icon} {result.test_case_id}: {result.test_case_name} "
                          f"[{result.status}] ({len(result.evidence)} evidence items)")

            except Exception as e:
                print(f"  ❌ ERROR in {suite_id}: {str(e)}")
                import traceback
                traceback.print_exc()

    def _build_auth_headers(self, har_analysis: HarAnalysis) -> dict:
        """Build authentication headers from HAR analysis."""
        headers = {}
        for token_type, token_value in har_analysis.auth_tokens.items():
            if 'bearer' in token_type.lower() or 'oauth' in token_type.lower():
                headers['Authorization'] = f"Bearer {token_value}"
            elif 'session' in token_type.lower():
                headers['X-SFDC-Session'] = token_value
                headers['Authorization'] = f"Bearer {token_value}"
        return headers

    def _generate_reports(self):
        """Generate HTML and JSON reports."""
        report_gen = ReportGenerator(
            output_dir=self.config['output']['report_dir'],
            evidence_dir=self.evidence_collector.run_dir
        )

        for fmt in self.config['output'].get('report_format', ['html']):
            if fmt == 'html':
                report_gen.generate_html_report(
                    self.all_results,
                    self.config['project'],
                    self.execution_start,
                    self.execution_end
                )
            elif fmt == 'json':
                report_gen.generate_json_report(
                    self.all_results,
                    self.config['project']
                )
```

```python
    def generate_json_report(self, results: list[TestResult], project_info: dict):
        """Generate a JSON report."""
        timestamp = datetime.utcnow().strftime("%Y%m%d_%H%M%S")
        filepath = self.output_dir / f"security_report_{timestamp}.json"

        report = {
            "project": project_info,
            "generated_at": datetime.utcnow().isoformat(),
            "summary": {
                "total_tests": len(results),
                "findings": sum(1 for r in results if r.status == "FINDING"),
                "not_findings": sum(1 for r in results if r.status == "NOT_FINDING"),
                "na": sum(1 for r in results if r.status == "NA"),
                "by_severity": {},
                "by_owasp_api": {},
                "by_owasp_web": {}
            },
            "results": [r.to_dict() for r in results]
        }

        for r in results:
            if r.status == "FINDING":
                report["summary"]["by_severity"][r.severity] = \
                    report["summary"]["by_severity"].get(r.severity, 0) + 1
                report["summary"]["by_owasp_api"][r.owasp_api_category] = \
                    report["summary"]["by_owasp_api"].get(r.owasp_api_category, 0) + 1

        with open(filepath, 'w', encoding='utf-8') as f:
            json.dump(report, f, indent=2, default=str)

        print(f"[Report] JSON report saved: {filepath}")
        return filepath
```

## Step 8: Main Entry Point

### `run_tests.py`
```python
#!/usr/bin/env python3
"""
Salesforce Portal API Security Testing Framework
=================================================

Main entry point for executing end-to-end security assessments
aligned with OWASP Top 10 Web, OWASP Top 10 API, and OWASP Secure Coding Guidelines.

Usage:
    python run_tests.py                          # Run full assessment
    python run_tests.py --target assist_portal   # Run for specific target
    python run_tests.py --suite API1-BOLA        # Run specific test suite
    python run_tests.py --list                   # List all available tests
"""

import sys
import argparse
from pathlib import Path
from rich.console import Console
from rich.table import Table
from rich.panel import Panel

# Add project root to path
sys.path.insert(0, str(Path(__file__).parent))

from src.test_engine import TestEngine

console = Console()


def print_banner():
    console.print(Panel.fit(
        "[bold cyan]🔒 Salesforce Portal API Security Testing Framework[/bold cyan]\n"
        "[dim]OWASP API Top 10 (2023) | OWASP Web Top 10 (2021) | Secure Coding Guidelines[/dim]",
        border_style="cyan"
    ))


def list_tests(engine: TestEngine):
    """Display all available test suites."""
    table = Table(title="Available Test Suites")
    table.add_column("Suite ID", style="cyan", no_wrap=True)
    table.add_column("Name", style="green")
    table.add_column("OWASP API", style="yellow")
    table.add_column("OWASP Web", style="magenta")

    for suite_id, test_class in engine.TEST_REGISTRY.items():
        table.add_row(
            suite_id,
            test_class.SUITE_NAME,
            getattr(test_class, 'OWASP_API', 'N/A'),
            getattr(test_class, 'OWASP_WEB', 'N/A')
        )

    console.print(table)


def main():
    parser = argparse.ArgumentParser(
        description="Salesforce Portal API Security Testing Framework"
    )
    parser.add_argument(
        "--config", default="config/settings.yaml",
        help="Path to configuration file (default: config/settings.yaml)"
    )
    parser.add_argument(
        "--target",
        help="Run tests for a specific target only (e.g., assist_portal)"
    )
    parser.add_argument(
        "--suite",
        help="Run a specific test suite only (e.g., API1-BOLA)"
    )
    parser.add_argument(
        "--list", action="store_true",
        help="List all available test suites and exit"
    )

    args = parser.parse_args()

    print_banner()

    engine = TestEngine(config_path=args.config)

    if args.list:
        list_tests(engine)
        return

    # Filter targets if specified
    if args.target:
        if args.target not in engine.config['targets']:
            console.print(f"[red]Error: Target '{args.target}' not found in config.[/red]")
            console.print(f"Available targets: {list(engine.config['targets'].keys())}")
            sys.exit(1)
        engine.config['targets'] = {args.target: engine.config['targets'][args.target]}

    # Filter suites if specified
    if args.suite:
        if args.suite not in engine.TEST_REGISTRY:
            console.print(f"[red]Error: Suite '{args.suite}' not found.[/red]")
            console.print(f"Available suites: {list(engine.TEST_REGISTRY.keys())}")
            sys.exit(1)
        engine.TEST_REGISTRY = {args.suite: engine.TEST_REGISTRY[args.suite]}

    # Run the assessment
    try:
        results = engine.run_full_assessment()

        # Print final summary table
        console.print("\n")
        table = Table(title="📋 Final Results Summary")
        table.add_column("ID", style="cyan")
        table.add_column("Test", style="white")
        table.add_column("Status", justify="center")
        table.add_column("Severity", justify="center")
        table.add_column("Evidence", justify="right")

        for r in results:
            status_style = {
                "FINDING": "[bold red]🔴 FINDING[/bold red]",
                "NOT_FINDING": "[green]🟢 PASS[/green]",
                "NA": "[dim]⚪ N/A[/dim]",
                "ERROR": "[yellow]🟡 ERROR[/yellow]"
            }.get(r.status, r.status)

            severity_style = {
                "CRITICAL": "[bold red]CRITICAL[/bold red]",
                "HIGH": "[red]HIGH[/red]",
                "MEDIUM": "[yellow]MEDIUM[/yellow]",
                "LOW": "[blue]LOW[/blue]",
                "INFO": "[dim]INFO[/dim]"
            }.get(r.severity, r.severity)

            table.add_row(
                r.test_case_id,
                r.test_case_name[:50],
                status_style,
                severity_style,
                str(len(r.evidence))
            )

        console.print(table)

    except KeyboardInterrupt:
        console.print("\n[yellow]Assessment interrupted by user.[/yellow]")
        sys.exit(1)
    except Exception as e:
        console.print(f"\n[red]Fatal error: {e}[/red]")
        import traceback
        traceback.print_exc()
        sys.exit(1)


if __name__ == "__main__":
    main()
```
